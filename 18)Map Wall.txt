//----- Green flag events ---------------------------------------------------------------------------------------------

WhenGreenFlagClicked()
{
    Motion.GoToXY(0, 0);
    Forever
    {
        Looks.SwitchCostumeTo(scene);
        Looks.SetEffectTo(GHOST, (G + 70));
        Looks.SetEffectTo(COLOR, -77);
    }
}


//----- Broadcast received events -------------------------------------------------------------------------------------

WhenBroadcastReceived(map)
{
    Looks.SetEffectTo(GHOST, 50);
    Repeat (4)
    {
        Looks.SetEffectTo(GHOST, 50);
    }
}
