//----- Green flag events ---------------------------------------------------------------------------------------------

WhenGreenFlagClicked() //Toggle minimap on and off script 1, and this one also handles costume switching.
{
    Motion.GoToXY(0, 0);
    Forever
    {
        Looks.SwitchCostumeTo(scene);
        Looks.SetEffectTo(GHOST, (G + 70));
        Looks.SetEffectTo(COLOR, -77);
    }
}


//----- Broadcast received events -------------------------------------------------------------------------------------

WhenBroadcastReceived(map) //Toggle minimap on and off script 2, and this one handles the delay.
{
    Looks.SetEffectTo(GHOST, 50);
    Repeat (4)
    {
        Looks.SetEffectTo(GHOST, 50);
    }
}
