//----- Green flag events ---------------------------------------------------------------------------------------------

WhenGreenFlagClicked() //Sets the HUD.
{
    Looks.SwitchBackdropTo(backdrop1);
    Looks.SetEffectTo(COLOR, 100);
    Level = 1;
    scene = 1;
    Exp pts = 0;
    Forever
    {
        Sound.PlayUntilDone(scene);
    }
}

WhenGreenFlagClicked()
{
    Call Full Rec;
    Used Potions = 0;
    Forever
    {
        If (Level == 20)
        {
            EXP DISPLAY = Operator.Join("L:20 E:320000 N:0", Operator.Join(" P:", (Potions - Used Potions)));
        }
        Else
        {
            EXP DISPLAY = Operator.Join(Operator.Join(Operator.Join("L:", Level), Operator.Join(" E:", Operator.Join(Exp pts, Operator.Join(" N:", (expreqs[Level] - Exp pts))))), Operator.Join(" P:", (Potions - Used Potions)));
        }
        If (Health < ((Level * 8) + 35))
        {
            Health += (((Level * 8) + 35) / 1000);
        }
        If (Health < 0)
        {
            Stop(all);
        }
        frame += 1;
    }
}

WhenGreenFlagClicked() //Healing potion usage
{
    Forever
    {
        If ((Sensing.KeyPressed(h) Or List.ContainsItem(mobileinputs, "heal")) And (Used Potions < Potions))
        {
            Sound.Play(Oops);
            Call Full Rec;
            Repeat (17)
            {
                Control.Wait(0);
                List.DeleteItem(mobileinputs, List.IndexOf(mobileinputs, "heal"));
            }
        }
    }
}

WhenGreenFlagClicked() //This controls unlocked weapons
{
    Forever
    {
        Weapon Limit = floor(((Level / 3.02) + 1));
    }
}


//----- Broadcast received events -------------------------------------------------------------------------------------

WhenBroadcastReceived(EnemyDefeated) //Level up script
{
    Repeat (20)
    {
        If (Exp pts > (expreqs[Level] - 1))
        {
            Sound.Play(Rank Up);
            Level += 1;
        }
        Else
        {
            Stop(this script);
        }
        Control.Wait(0);
    }
}
 
WhenBroadcastReceived(next) //Change color of empty space to better reflect environment
{
    If (scene == 2)
    {
        Looks.SetEffectTo(COLOR, 60);
    }
    If (scene == 3)
    {
        Looks.SwitchBackdropTo(backdrop2);
    }
    If (scene == 4)
    {
        Looks.SetEffectTo(BRIGHTNESS, 15);
    }
    If (scene == 5)
    {
        Looks.SwitchBackdropTo(backdrop1);
        Looks.SetEffectTo(COLOR, -30);
        Looks.SetEffectTo(BRIGHTNESS, 0);
    }
    If (scene == 6)
    {
        Looks.SetEffectTo(COLOR, 0);
    }
    If (scene == 7)
    {
        Looks.SetEffectTo(BRIGHTNESS, -30);
    }
    If (scene == 8)
    {
        Looks.SetEffectTo(BRIGHTNESS, 20);
        Looks.SetEffectTo(COLOR, 60);
    }
    If (scene == 9)
    {
        Looks.SetEffectTo(COLOR, 30);
    }
}

WhenBroadcastReceived(next) //Recovery
{
    Call Full Rec;
    Used Potions += -1;
}

{
    Health = ((Level * 8) + 35);
    Used Potions += 1;
