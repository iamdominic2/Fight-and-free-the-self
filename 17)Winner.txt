//----- Green flag events ---------------------------------------------------------------------------------------------

WhenGreenFlagClicked()
{
    w1 = 0;
    w2 = 0;
    Motion.GoToXY(0, 0);
    Looks.Hide();
    Costume.GoToFrontBack(front);
    Wait Until ((w1 == 1) And (w2 == 1));
    Looks.Show();
}


//----- Broadcast received events -------------------------------------------------------------------------------------

WhenBroadcastReceived(Win1)
{
    w2 = 1;
}

WhenBroadcastReceived(Win2)
{
    w1 = 1;
}
